import sys
import random
import collections
import matplotlib.pyplot as plt


class Poisson:
    def __init__(self, rate=sys.argv[1]):
        """init"""
        self.rate = int(rate)

        self.time = 1  # å•ä½æ—¶é—´
        if len(sys.argv) > 2 and sys.argv[2] != "":
            self.time = int(sys.argv[2])  # æ‰‹åŠ¨æŒ‡å®šæ—¶é—´èŒƒå›´

        self.EXP_NUM = 100000  # å®éªŒæ¬¡æ•°
        self.NUM_LEVEL = 2  # æ•°é‡çº§

    def generator(self, prob):
        """ä»¿çœŸç»“æœç”Ÿæˆå™¨"""
        while True:
            if random.random() < prob:
                yield 1
            else:
                yield 0

    def conduct_one_experiment(self, rate, time):
        """è¿›è¡Œä¸€æ¬¡å®éªŒ
           æ¯æ¬¡å®éªŒä¸­ï¼Œæ—¶é—´åˆ†ç‰‡çš„æ•°é‡æ¯”rateé«˜ä¸¤ä¸ªæ•°é‡çº§
           return: æœ¬æ¬¡å®éªŒä¸­ï¼Œäº‹ä»¶å‘ç”Ÿçš„æ¬¡æ•°
        """
        level = len(str(rate))
        shard_num = 10 ** (level + self.NUM_LEVEL)  # è®¡ç®—æ—¶é—´åˆ†ç‰‡çš„æ•°é‡

        gen = self.generator(rate / shard_num)

        cnt = 0
        for _ in range(time * shard_num):
            cnt += next(gen)
        return cnt

    def get_distribution(self, exp_num, rate, time):
        """å¤šæ¬¡å®éªŒï¼Œå¾—åˆ°åˆ†å¸ƒ
           exp_num: å®éªŒæ¬¡æ•°
        """
        lst = []
        for _ in range(exp_num):
            lst.append(self.conduct_one_experiment(rate, time))

        return sorted(collections.Counter(lst).items(), key=lambda e: e[0])

    def draw(self, sorted_list):
        """ç”»å›¾"""
        s = sum([e[1] for e in sorted_list])
        x = [e[0] for e in sorted_list]
        y = [e[1] / s for e in sorted_list]

        plt.plot(x, y)
        
        plt.xlabel("k")
        plt.ylabel("P(k)")
        plt.show()

    def main(self):
        """main"""
        sorted_list = self.get_distribution(self.EXP_NUM, self.rate, self.time)
        self.draw(sorted_list)

    @staticmethod
    def calculator(rate, t, k):
        """ç”¨äºè®¡ç®—æ³Šæ¾å‡½æ•°çš„æ¦‚ç‡ P(k|t,lambda)
           rate: lambda
           t: t
           k: k
        """
        import math
        return (rate * t) ** k / math.factorial(k) * math.exp(-rate * t)
