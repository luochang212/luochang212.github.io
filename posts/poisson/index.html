class Poisson:

        """init"""
        self.rate = int(rate)

        self.time = 1  # å•ä½æ—¶é—´
        if len(sys.argv) > 2 and sys.argv[2] != "":
            self.time = int(sys.argv[2])  # æ‰‹åŠ¨æŒ‡å®šæ—¶é—´èŒƒå›´

        self.EXP_NUM = 100000  # å®éªŒæ¬¡æ•°
        self.NUM_LEVEL = 2  # æ•°é‡çº§

    def generator(self, prob):
        """ä»¿çœŸç»“æœç”Ÿæˆå™¨"""
        while True:

    def conduct_one_experiment(self, rate, time):
        """è¿›è¡Œä¸€æ¬¡å®éªŒ
           æ¯æ¬¡å®éªŒä¸­ï¼Œæ—¶é—´åˆ†ç‰‡çš„æ•°é‡æ¯”rateé«˜ä¸¤ä¸ªæ•°é‡çº§

    def get_distribution(self, exp_num, rate, time):
        """å¤šæ¬¡å®éªŒï¼Œå¾—åˆ°åˆ†å¸ƒ
           exp_num: å®éªŒæ¬¡æ•°

    def draw(self, sorted_list):

        s = sum([e[1] for e in sorted_list])
        x = [e[0] for e in sorted_list]
